using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace C_Sharp_Practice
{
    class GenericList<T>
    {
        List<T> myList = new List<T>();

        public void Add(T t)
        {
            myList.Add(t);
        }
        public T Get(int i)
        {
            return myList[i];
        }
    }

    class ImplementInt
    {
        static GenericList<int> gl = new GenericList<int>();

        public static void fillList()
        {
            for(int ii=0; ii<7; ++ii)
            {
                gl.Add(ii);
            }
        }
        public static void printList()
        {
            for(int ii=0; ii<7; ++ii)
            {
                Console.WriteLine(gl.Get(ii));
            }
        }
    }

    class Practice_Problem
    {

        public static void Practice_Problem_Main()
        {
            ImplementInt.fillList();
            ImplementInt.printList();
        }
    }
}
